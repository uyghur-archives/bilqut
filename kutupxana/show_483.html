  بۈگۈن تور بەت ئارلاۋىتىپ ، مەن ئەسىرىنى ياقتۇرۇپ ئوقۇيدىغان توردىشىم ئەزىز ئىمىر يانغىنىڭ يەنە بىر ياخشى يېزىلغان ئەسىرى <<مەن پاھىشە ئەمەس >> ناملىق ئەسىرىنى كۈرۈپ قالدىم . ئۇقۇپ ئۈزەممۇ بىلمەي تەسىرلىنىپ ئەسەرنىڭ ئىچىگە كىرىپ كىتىپتىمەن . مەن بىلىقۇت شىركىتىنىڭ بۇ ئەسەرنى ئاۋازغا ئېلىپ چىقىپ ، جەمئىيتىمىزدىكى ئازغۇن ياشلارغا بىر ياخشى سىگىنال بىرىشىنى ئۈمۈد قىلىمەن . ئۇلۇغ ئاللاھ قىلغان ئەمگىكىڭلارغا چۇشلۇق سىلەرگە تېخىمۇ ئۇتۇق ، بەرىكەتلەرنى ئاتا قىلسۇن .... بىلىقۇت شىركىتىنىڭ يېقىندىن بۇيان توردىكى نادىر ئەسەرلەرنى ، ياخشى رومانلارنى ئاۋازغا ئېلىپ كەلگىنىگە يەنە بىر قېتىم تەشەككۈر !!! [align=center][color=#FF0000]مەن ئەسلى پاھىشە ئەمەس (ھىكايە ) ئاپتۇرى : ئەزىز ئىمىر يانغىن [/color][/align] مەن ئەسلى پاھىشە ئەمەس ، مىنىڭمۇ ئاجايىپ تاتلىق ، ئاجايىپ خۇشال ۋاقىتلىرىم بولغان . مىنىڭمۇ نۇرغۇن ئارزۇ _ئارمانلىرىم ،غايىلىرىم بولغان . ئەمما ئاشۇ بىر قېتىملىق خاتا ئالغان بىر قەدەم ، ئاشۇ بىز تەرەپلەپ كىلىۋاتقان <<ئۇلۇغ ، قىممەتلىك ،پاك >>مۇھاببەت مىنىڭ بۇ يولغا كىرىپ قېلىشىمغا سەۋەپچى بولغان .مىنىڭ تۇنجى سۆيگەن يىگىتىم ،تاغدەك ئىشەنگەن ئادىمىم ، ئاغزىدىن <<دىن ، مىللەت >> بىلىمىنى چۈشۈرمەيدىغان ئاشۇ يىگىت مىنى بۇ يولغا باشلاپ كىرگەن .ئىشەنمىسىڭىز مىنىڭ بۇ ھىكايەمنى ، ياق مىنىڭ بۇ يۈرىكىمنى پارە -پارە قىلغان ، قانلىرىمنى يىرىڭ قىلغان ،بۇ قىسمەتلىك تەقدىرىمنى ئاڭلاپ بېقىڭ ..                  [align=center]ئۆتكۈزگەچ چوڭ گۇناھ ، يۈرىكىم پارە ، قانلىرىم يىرىڭ بۇلۇپ ، بار جىسمىم بولدى يارە . گۆشلىرىم سېسىپ مىنىڭ ،سۆڭەكلىرىم كەتتى چىرىپ ، كۆز يېشىم دەريا بولدى ،ئەمدى بۇنىڭغا نە چارە [/align] مىنىڭ ئالىي مەكتەپتە ئوقىۋاتقان مەزگىللىرىم ئىدى ،بىز ئىككىمىزنىڭ يۈرگىنىنى مەكتەپتىكى ھەممە بىلەتتى ، ھەممەيلەن بىزگە ھەۋەس بىلەن قارايتتى .تېخى بىزنى مەكتەپنىڭ <<تىتانىكلىرى >> دەپ ئاتىشاتتى .مەن ئۇنى شۇنچىلىك ياخشى كۈرەتتىمكى ،ئۇنىڭ ئۈچۈن ھەممە نەرسەمنى قۇربان قىلىش ئۈچۈن تەييار ئىدىم .مۇھاببىتىمىز ئۈچۈن ئىچكەن قەسەملىرىمىزنى باشقا قىزلارغا كۆپتۈرۈپ سۆزلەپ كۆز _كۆز قىلاتتىم .ئاخىرى ئوقۇش پۈتتۈردىغان ۋاقتىمىز كەلدى ، مەكتەپ پۈتتۈردىغان ئەڭ ئاخىرقى بىر كۈنىمىز قالدى .مىنىڭ بۇ قارا قىسمەتلىرىمنىڭ باشلىنىشىغىمۇ بىر كۈن قالدى .بىز ئىككەيلەن شۇ ئەڭ ئاخىرقى بىر كۈن ئاخشىمى بىر _بىرىمىزگە قانماي مۇڭداشتۇق ، يىغلاشتۇق .نەچچە مىڭ قېتىم بىر-بىرىمىزدىن ۋاز كەچمەسلىك توغۇرلۇق قەسەم ئىچىشتۇق . مەن ئاشۇ قەسەملەرگە ،ۋەدىلەرگە ، يەنە شۇ ئىپلاسنىڭ يالۋۇرۇشلىرىغا ئىشىنىپ ھاياتىمدىكى ئەڭ چوڭ خاتالىقنى ئۆتكۈزدۈم .بىز قىزلارنىڭ ھاياتىدىنمۇ قىممەتلىك بولغان نەرسەمدىن ئايرىلدىم ....     مانا ھەممەيلەن يۇرتىمىزغا قايتىپ كەتتۇق . مىنىڭ ئۆيۈمگە كىلىپ قىلغان بىرىنچى ئىشىم ئۇنىڭدىن تېلفۇن كۈتۈش بولدى .ئۇنىڭدىن بىر ئايغىچە تېلفۇن ۋە خەت كىلىپ تۇردى .بىر ئايدىن كىيىن ئۇنىڭدىن خەتتىن تۈگۈل بىرەر قېتىم تېلفۇنمۇ كەلمىدى .قاتتىق تىت -تىت بۇلۇشقا باشلىدىم .ئارىدىن ئون كۈن ئۆتكەندە بىر كېچىسى ئۇنىڭدىن تېلفۇن كەلدى .ئۇ يىغلامسىرغان ھالدا ئانىسىنىڭ ئىككەيلەننىڭ ئىشىغا قۇشۇلمايدىغانلىقىنى ،ئانىسىنى نارازى قىلغۇسى يوقلىغىنى ،ئەمدى تېلفۇن قىلالمايدىغانلىقىنى ئېيتىپ تېلفۇننى قۇيىۋەتتى . ھۇشۇمدىن كەتكىلى تاسلا قالدىم ،بىر كېچە ئۆينىڭ ئارقىسىغا چىقىپ توختىماي يىغلاپ چىقتىم .ئۆگەي ئانامنىڭ <<زەھەردىن قاتتىق >>سۆزلىرىنى ئاڭلىغاندىن كىيىن ،تاڭ ئاتارغا يېقىن ئۆيگە كىرىپ كەتتىم . شۇنىڭدىن كىيىن ئۇنىڭغا ھەرقانچە تېلفۇن قىلىپمۇ ئالالمىدىم .نەچچە يۈز پارچە خەتمۇ يازدىم .ئەڭ ئاچچىقىمنى كەلتۈرگىنى ، ئۇ ئۆيىنىڭ تېلفۇن نۇمۇرىنى ئالماشتۇرۋىتىپتۇ . قانداق قىلاي ؟؟؟ ئۇنىڭ قېشىغا بېرىپ يالۋۇرايمۇ ؟؟؟بىراق ، مەسچىتتىن ، نامازدىن باشقا يەرگە بارمايدىغان دادام ، گەپلىرى ھىڭدىن سېسىق ئۆگەي ئانام مىنى يالغۇز خوتەندىن ئۈرۈمچىگە يولغا سېلىپ قۇيارمۇ ؟؟؟....     ئارىدىن نەچچە ئاي ئۆتتى ، مەن ئۈزەمدىكى ئۆزگىرىشلەرنى كۈرۈپ ،قورقۇپ ئۆلەيلا دىدىم . ئاخىرى بىر خەنزۇ ئايال ئاچقان دوختۇرخانىغا كىرىپ ئۈزۈمنى تەكشۈرتتۈم . نالەت ساڭا رازىيە ، نالەت ! ئاللاھ سىنى جازالىدى ،ئاھ ، خۇدا ئەمدى قانداقمۇ قىلارمەن ؟ئىلگىرى كىنولاردىن ، كىتاپلاردىن بۇنداق قىزلارنىڭ ۋەقەلىگىنى كۆرسەم ، ئۇ قىزغا ئېچىنىپ كىتەتتىم ، لىكىن بۇ كۈن مىنىڭ بېشىمغا كەلدى . مىنىڭ بۇيۇمدا قاپتۇ ، نىمە نۇمۇس بۇ ؟ نىمە ئات بۇ ؟!!...    ئەمدى قانداق قىلاي ؟كىم ماڭا ياردەم قىلار ؟ ئىلگىرى ماڭا تاغدەك يۆلەنچۈك  بىلىنگەن شۇ نامەرت ، لاتىنى ئىزدەيمۇ ؟ياكى مىنى دادامغا چېقىشتۇرۇشتىن باشقىنى بىلمەيدىغان چايان تىللىق ئۆگەي ئانامدىن ياردەم ئىزدەيمۇ ؟؟ ياكى بۇ دۇنيادا يەنە ياشاشقا شەرتى توشمايدىغان بۇ ناپاك تىنىمنى ئېلىپ ،بۇ دۇنيادىن مەڭگۈلۈك خوشلىشايمۇ ؟؟...مەن كىيىنكىسىنى تاللاشنى قارار قىلدىم ،بىراق ، ئادەمگە ئەمىلىيەتتە ئەڭ تەس بولغىنى ئۆز جېنىدىن ۋاز كىچىش ئىكەن .    دەردىمنى ئىچىمدە بىلىپ ، كىچىلىرى يىغلاپ _قاقشاپ كۈندىن _كۈنگە يوغىناۋاتقان قورسىقىمنى مەھكەم تېڭىۋىلىپ يۈرىۋەردىم . مانا ئاي _كۈنلەر ئۈتۈپ ، بىر كۈنى قورسىقىمدىكى بۇ گۇناھىمنىڭ جازاسى بولغان نارسىدە تۇغۇلۇش ئۈچۈن ھەركەت قىلىشقا باشلىدى . قورققىنىمدىن ئۆيدىن شۇنچە تېز يۈگۈرۈپ چىقتىمكى ، ئادەم بارمايدىغان چەت ئېتىزلىقتىكى قىغ دۆۋىسىنىڭ كەينىگە بېرىپ يۇقۇلىنىشقا باشلىدىم .بىراق ،بىرەر سائەت ئۈتۈپمۇ بۇ نارسىدىنىڭ بۇ دۇنياغا كۆز ئاچقۇسى كەلمىدى ، يا مىنىڭ جېنىم چىقمىدى . ئاخىرى ئاغرىققا چىدىماي ، ئالدىنئالا بۇ ئىش ئۈچۈن ناھايىتى تەستە يىققان 100 يۈ ەن پۇلنى ئېلىپ ماي يولىغا چىقىپ كىچىك ماشىندىن بىرنى توستۇم . ھەم 50يۈ ەنگە شەھەرلىك دوختۇرخانىنىڭ ئالدىغا ئاپىرىپ قۇيۇشقا دىيىشتىم . ماشىندا ئۈنلۈكرەك ۋارقىراپ سالسام شوپۇرنىڭ گۇمانلىنىپ قېلىشىدىن ئەنسىرەپ لىۋىمنى چىشلەپ قان چۇماق قىلىۋەتتىم . مانا دوختۇرخانىغىمۇ كىردىم ، بىراق تۇيۇقسىز تۇغۇت بۈلۈمىنىڭ ئالدىدا تۇرغان قوشنىمىزنى كۈرۈپ قالدىم  ، يەنە ئامالسىز بۇ دوختۇرخانىدىن يېنىپ چىقىپ ، كىچىك ماشىندىن بىرنى كىرا قىلىپ ، شەھەرلىك 2-دوختۇرخانىنىڭ ئالدىغا كەلدىم . بىراق ماڭىدىغانغا پۇت _قۇلۇمدا ئاغرىق دەستىدىن جان دىدارى قالمىغانىدى ، ئۆمىلەپ دىگۈدەك دوختۇرخانىنىڭ تۇغۇت بۈلۈمىگە چىقىپ ، كارۋىتىغا ئۈزۈمنى تاشلىدىم . بىر ئايال دوختۇر مىنى تەكشۈرۈپ كۈرۈپ ، دەرھال ئوپراتسىيە قىلمىسا بولمايدىغانلىقىنى ، دەرھال پۇل ئۆتكۈزۈپ ، ئۇرۇق _تۇققانلىرىمدىن بىرەرسىنىڭ قول قۇيۇپ بىرىشىنىڭ لازىملىغىنى ئېيتتى . مەن دوختۇرغا :<<مەن يالغۇز ، مىنىڭ ھېچقانداق ئۇرۇق _تۇققىنىم ، پۇلۇم يوق .>>دىدىم .بۇ گىپىمنى ئاڭلىغان دوختۇر ئايال  ھەم باشقا كىشىلەر ماڭا مەنسىتمەسلىك بىلەن قاراپ كەتتى ھەم دوختۇر ئايال "<< ئۆلگۈڭ كەلگەن نىمە ئوخشايسەن .>>دەپ يېنىدىكى ئىككەيلەنگە :<<بۇنىڭ پۇلى بولمىسا باشقا شەخسىي دوختۇرخانىغا بارسۇن ، بىزدە ئامال يوق ، سىرتقا يۆلەپ ئېچىقىپ قۇيۇڭلار ، بۇ يەردە يېتىپ بىرەر ئىش بولۇپ قالسا ، بىزگە مەسئۇلىيەت كىلىپ قالىدۇ .>>دېدى . شۇنىڭ بىلەن ئۇ ئىككەيلەن مىنىڭ قولتۇقىمدىن تارتىپ دىگەندەك پەسكە چۈشۈرۋەتتى . مەن پەستە بىردەم پۇچلىنىپ ياتقاندىن كىيىن ، يەنە ئامالسىز شۇ بۈلۈمگە چىقتىم . پەلەمپەيدىن يامىشىپ چىقالماي ، قوللىرىمنىڭ بارماقلىرىمۇ ئۈزۈلۈپ كىتەيلا دەپ قالدى .ھەيرانمەن ، ئادەمنىڭ جېنى بەك قىينىلىپ كەتسە بۇرنىدىنمۇ ، كۈزىدىنمۇ قان ئاقىدىكەن ،  يۈزلىرىم قانغا بۇيىلىپ كەتكەن ئىدى . چىقىپلا خۇددى شۇ كارۋاتتا ياتسام ئاغرىق توختاپ قالىدىغاندەك ، تۇغۇت بۈلۈمىنىڭ كارۋىتىدا يېتىۋالدىم . بىراق ، بۇ قېتىم ئۇ دوختۇر ئايال ماڭا يۈز قىلىپ ئولتۇرمىدى ، ئىشخانىسىدىن ئېتىلىپ چىقىپ ، مىنىڭ ياقامدىن تۇتۇپ سىلكىپ قۇپۇردى ھەم كاچىتىمغا بىرنى سېلىپ : << قانداق بىر نىمە سەن ؟ ئېيتە ، ئۇرۇق _تۇققانلىرىڭمۇ يوقمۇ ؟ تېز ئېيت ، جان لازىم بولسا تېز دە ! بىزگە خاپىلىق تېپىپ بىرىدىغان ئوخشايسەن ..>>دەپ قىستاپ كەتتى . مەن گەپ قىلماي تۇرىۋالسەم مىنى يەنە پەسكە چۈشۈرۋەتتى .مەن بۇ قېتىم يىغلاپ خۇدادىن ئۈلۈمىمنى تىلەشكە باشلىدىم ، چاچلىرىمنى  ئۈزۈم يۇلدۇم . ئاخىرى مىنىڭ بۇ ھالىتىمگە چىدىمىغان 40ياشلار چامىسىدىكى بىر ئايال يىغلاپ تۇرۇپ ، بېشىمنى قۇچىقىغا ئېلىپ تۇرۇپ :<<جېنىم قىزىم ، بۇنداق قىلغۇچە راستنى دەڭ ، مەن ئاتا-ئانىڭىزغا خەۋەر قىلاي .>،دېدى . مەن يىغلاپ تۇرۇپ قەتئىي ئېيتالمايدىغانلىقىمنى ۋە مىنى قۇتقۇزۋىلىشىنى ئۈتۈندۈم . ئاخىرى ئۇ ئاقكۆڭۈل ئايال مىنى يۆلەپ تۇغۇت بۈلۈمىگە ئېلىپ كىردى ھەمدى ئۈزى پۇل تۆلىدى ، ئۆزى ئانامنىڭ ئورنىدا ئىمزا قۇيۇپ ئوپراتسىيىگە قول قويدى . ئوپراتسىيەدە ھېلىمۇ مەن ھايات  قاپتىمەن ، بالامنىمۇ ساق ئاپتۇ . ھېلىقى ئايال ماڭا بالامنى كۆرسىتىپ تۇرۇپ:<<نىمىدىگەن چىرايلىق ،  ھەددىدىن ئاتۇق گۈزەل چىرايى سىزگە ئوخشايدىكەن ، بەك چىرايلىق بولسىمۇ دۈشمەنلىك بۇلۇپ قالىدۇ ، ئىسمىنى "تۇمارخان " قۇيايلى . >> دېدى . ئۇ ئايال خۇددى ئۆز ئانامغا ئوخشاش ھالىمدىن بەك ياخشى خەۋەر ئالدى . توۋا قۇدرىتى ئۇلۇغ ئاللاھ ، مىھرى شەپقەتلىك ئاللاھ مىنى شۇ پەرىشتە سۈپەتلىك ئايالغا يۇلۇقتۇرغىنىڭغا رەھمەت .  ئارىدىن تۆت كۈن ئۆتكەندىن كىيىن مەن دوختۇرنىڭ تىللاشلىرىغا ، ھېلىقى ئايالنىڭ يالۋۇرۇشلىرىغا قارىماي دوختۇرخانىدىن چىقتىم . ھەمدە ئۇ ئايالغا <<ئۆمۈر بۇيى ئىزدىمەيمەن ...>>دەپ بالامنى بىرىۋەتتىم . ئۇ ئايالمۇ بەك خوش بۇلۇپ كەتتى .       مانا ئۆيگىمۇ كەلدىم . مىنىڭ تۆت كۈن سەۋەپسىز يوقاپ كەتكەنلىكىم ئۈچۈن دادام بىر كۈن ئۆيگە ئېسىپ قويدى . ئوپراتسىيە قىلغاندا كەسكەن يەردىن توختىماي قىزىل ئارلاش سۇ چىقاتتى . بىراق بۇنى دادامغا ئېيتالمايتتىم . بىر ئايدىن كىيىن مىنى مەجبۇرى ھالدا مەھەللىدىكى بىرسىگە  ئەرگە بىرىۋەتتى .  بىراق توي كىچىسى ئېرىم مىنىڭ ئەدەپ-ئەخلاقلىق ، ئىپپەتلىك قىز ئەمەسلىگىمنى بىلىپ ، يالۋۇرۇشلىرىمغا قارىماي بىر كېچە تاسما بىلەن ساۋاپ چىقتى ھەمدە ئەتىسى ئەتىگەندە پۈتۈن تويلۇقنى ياندۇردىغانلىقىنى ئېيتىپ يولغا سېلىپ قويدى . بىراق ،مەن ئۆيگە قايتالمايتتىم ، مەن يۇرتۇمدىن قېچىپ چىققاندىن كىيىن پۈتۈن ئوغۇللارغا ، پۈتۈن ئەركىشىگە قورقۇنچ بىلەن قارايدىغان بۇلۇپ قالدىم . ئاخىرى پاھىشلىكتىن ئىبارەت بۇ تۇيۇق يولغا كىرىپ قالدىم ....كۆز يېشىم تولا يىغلاپ قۇرۇپ كەتتى . بىلىمەن ، مىنى بۇ زىمىن ، ئاللاھ ھەرگىز كەچۈرمەيدۇ !      پەقەت مىنىڭ دەيدىغىنىم ئوغۇللارغا :<<قىزلارنى ئالدىماڭلار ، ئۇلارنى نابۇت قىلماڭلار ! >> قىزلارغا دەيدىغىنىم :<< مەكتەپتىكى ئاشۇ قۇرۇق مۇھاببەتنى دەپ ، بىردەملىك ھىسىياتنىڭ قۇلى بۇلۇپ مەندەك قىسمەتكە دۇچار بولماڭلار ! >>  مەن ئەسلى پاھىشە ئەمەس. پاھىشە ئەمەس ! ...... [color=#FF0000]يانغىن : [/color]   ھەممىمىز ئويلىنىپ باقايلى ، جەمئىيەتنىڭ ئۇچقاندەك تەرەققى قىلىشىغا ئەگىشىپ ، بىزنىمىلەرگە ئىرىشتۇق ، نىمىلەرنى يوقىتىۋاتىمىز ؟؟ !! ھايات مۇساپىمىزدە قانداق قىسمەتلەرگە ئۇچرايمىز ؟  ھاياتىمىزنى قانداق ئۆتكۈزۈشىمىز كىرەك ؟  بۇ ھەقتە پىكىر يۈرگۈزۈپ باقايلى !  ئاشۇ بىزلەر تەرىپىدىن << ئوغرى ، يانچۇقچى .. پاھىشە ..>> ۋە ئالىقانداق سەت ناملار بىلەن ئاتىلىپ كىلىۋاتقان شۇلارنى كۆز ئالدىمىزغا كەلتۈرۈپ باقايلى !  مەن ئويلايمەن ، ھېچكىم تۇغۇلىشىدىنلا ئوغرى ، پاھىشە بۇلۇپ يارالمايدۇ ، ھېچكىمنىڭ شۇ ئىشنى ئۈزى ئۈمۈت قىلىپ ، خالاپ قىلغۇسى يوق !   شۇ ئىشنى قىلىشقا زورلاۋاتقانلار يەنىلا بىز ھەم ئەتراپىمىزدىكى كىشىلەر !  ...ئاشۇنداق ناملار بىلەن ئاتالغانلارنىڭمۇ باشقىلارنىڭ ھۆرمىتىگە ئىرىشىپ ، ياخشى خىزمەت قىلىپ ياشىغۇسى ، ھاياتىنى ئەھمىيەتلىك ، گۈزەل ئۆتكۈزگۈسى كىلىدۇ ، لىكىن شارائىت يوق !  ... بىز ئۇلارغا شارئىت يارىتىپ بەرمەيمىز !  ئۇلارنى ئالىقانداق سەت ناملار بىلەن تىللايمىزيۇ ، ئۆزىمىز ئۈستىدە ئويلىنىپ باقمايمىز ! بۇ ئەسەرنى يوللاپ ياشلىرىمىز ، جەمئىيىتىمىزنىڭ ھەر قايسى قاتلاملىرىدا ئىشلەۋاتقان ھەر تۈرلۈك كىشىلىرىمىزنىڭ كاللىسدا ئويلىنىش ،مۇلاھىزە پەيدا قىلالىسام مەن بۇ ئەسەرنى يوللىغىنىمدىن خۇشال بولغان بۇلاتتىم .
